function colorFor(v){return v>=70?'#2f9e6b':v>=50?'#dd9a34':'#d1564c';}
function computeScore(ind,w){ const tw=w.reduce((a,b)=>a+b,0)||1; return Math.round(ind.reduce((s,v,i)=>s+v*w[i],0)/tw); }
function confidenceOf(d){ const n=SOURCES.length; const have=SOURCES.filter(s=>d.dataAvail[s]).length; return Math.round(have/n*100); }
function missingSources(d){ return SOURCES.filter(s=>!d.dataAvail[s]); }

/* ---------- weights / settings ---------- */
function onWeight(){
  let vals=[]; for(let i=0;i<5;i++){ const v=+document.getElementById('w'+i).value; vals.push(v); document.getElementById('wv'+i).textContent=v+'٪'; }
  const sum=vals.reduce((a,b)=>a+b,0);
  const el=document.getElementById('wsum');
  el.textContent = 'المجموع الحالي: '+sum+'٪'+(sum===100?' ✓':' — يجب أن يساوي ١٠٠٪');
  el.className = 'wsum '+(sum===100?'ok':'bad');
}
function applyWeights(){
  let vals=[]; for(let i=0;i<5;i++) vals.push(+document.getElementById('w'+i).value);
  const sum=vals.reduce((a,b)=>a+b,0);
  if(sum!==100){ alert('مجموع الأوزان يجب أن يكون ١٠٠٪ بالضبط قبل التطبيق.'); return; }
  WEIGHTS=vals;
  renderCompany(currentCR);
  showTab('dash');
}
function resetWeights(){
  DEFAULT_WEIGHTS.forEach((v,i)=>{ document.getElementById('w'+i).value=v; document.getElementById('wv'+i).textContent=v+'٪'; });
  onWeight();
}

function sparkSVG(hist){
  const w=200,h=40,max=Math.max(...hist),min=Math.min(...hist),range=(max-min)||1;
  const pts=hist.map((v,i)=>{ const x=(i/(hist.length-1))*w; const y=h-((v-min)/range)*h; return x+','+y; }).join(' ');
  const last=hist[hist.length-1];
  return `<svg width="${w}" height="${h}" viewBox="0 0 ${w} ${h}"><polyline points="${pts}" fill="none" stroke="${colorFor(last)}" stroke-width="2.5" stroke-linecap="round" stroke-linejoin="round"/></svg>`;
}

function toggleWhy(){ whyOpen=!whyOpen; document.getElementById('whyBox').classList.toggle('open',whyOpen); }
function buildWhy(d,score){
  const box=document.getElementById('whyBox'); box.classList.toggle('open',whyOpen);
  const tw=WEIGHTS.reduce((a,b)=>a+b,0)||1;
  let rows='';
  d.ind.forEach((v,i)=>{
    const contrib = (v*WEIGHTS[i]/tw);
    rows += `<div class="row"><span>${IND[i]} (وزن ${WEIGHTS[i]}٪)</span><b>${contrib.toFixed(1)} نقطة</b></div>`;
  });
  box.innerHTML = rows + `<div class="total"><span>مجموع الدرجة</span><span>${score}</span></div>`;
}

/* ---------- فرص التحسين: weakest indicators -> immediate action items ---------- */
function buildOpportunities(d, score){
  const box=document.getElementById('opps');
  const target = Math.max(75, d.sectorAvg);
  const items = d.ind
    .map((v,i)=>({i,v,name:IND[i],tip:TIPS[i]}))
    .filter(x=>x.v<70)
    .sort((a,b)=>a.v-b.v);

  if(items.length===0){
    box.innerHTML = `<div class="oppgood">✓ لا توجد نقاط ضعف جوهرية حالياً — جميع المؤشرات ضمن نطاق آمن</div>`;
    return;
  }

  box.innerHTML = items.map(it=>{
    const priority = it.v<40 ? {label:'أولوية عاجلة',color:'#d1564c'} : it.v<60 ? {label:'أولوية متوسطة',color:'#dd9a34'} : {label:'تحسين إضافي',color:'#8785d8'};
    const hyp = d.ind.map((v,idx)=> idx===it.i ? target : v);
    const newScore = computeScore(hyp, WEIGHTS);
    const gain = Math.max(0, newScore-score);
    return `
      <div class="oppitem">
        <span class="pr" style="background:${priority.color}">${priority.label}</span>
        <div class="body">
          <div class="name">${it.name} — القيمة الحالية ${it.v}</div>
          <div class="tip">${it.tip}</div>
          <div class="metrics"><span>الهدف: <b>${target}</b></span><span>الأثر المتوقع على الدرجة: <b>+${gain}</b> نقطة</span></div>
        </div>
      </div>`;
  }).join('');
}
